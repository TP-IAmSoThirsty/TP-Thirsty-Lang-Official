 45class MutationParser:
 46    """Parses mutation definitions from source text."""
 47
 48    MUTATION_RE = re.compile(
 49        r'mutation\s+(\w+)\s*\{\s*'
 50        r'validated_canonical\s*\{'
 51        r'(.*)\}'
 52        r'\s*\}',
 53        re.DOTALL
 54    )
 55
 56    BLOCK_RE = re.compile(
 57        r'(shadow|invariant|canonical)\s*\{'
 58        r'(.*?)\}',
 59        re.DOTALL
 60    )
 61
 62    @classmethod
 63    def parse(cls, text: str) -> ShadowModule:
 64        """Parse mutation source text into a ShadowModule."""
 65        text = text.strip()
 66        mut_match = cls.MUTATION_RE.search(text)
 67        if not mut_match:
 68            raise ValueError("No valid mutation definition found")
 69
 70        name = mut_match.group(1)
 71        validated_canonical_body = mut_match.group(2)
 72
 73        module = ShadowModule(name=name, source=text)
 74        shadow_code = ""
 75        invariant_code = ""
 76        canonical_code = ""
 77
 78        for block_match in cls.BLOCK_RE.finditer(validated_canonical_body):
 79            block_type = block_match.group(1).strip()
 80            block_content = block_match.group(2).strip()
 81
 82            if block_type == 'shadow':
 83                shadow_code = block_content
 84                module.shadow_code = shadow_code
 85            elif block_type == 'invariant':
 86                invariant_code = block_content
 87                module.invariant_code = invariant_code
 88            elif block_type == 'canonical':
 89                canonical_code = block_content
 90                module.canonical_code = canonical_code
 91
 92        # Also try parsing without the nested validated_canonical wrapper
 93        if not shadow_code and not invariant_code and not canonical_code:
 94            direct_blocks = cls.BLOCK_RE.finditer(text[len(f"mutation {name}"):])
 95            for block_match in direct_blocks:
 96                block_type = block_match.group(1).strip()
 97                block_content = block_match.group(2).strip()
 98                if block_type == 'shadow':
 99                    module.shadow_code = block_content
100                elif block_type == 'invariant':
101                    module.invariant_code = block_content
102                elif block_type == 'canonical':
103                    module.canonical_code = block_content
104
105        return module

110class PlaneIsolationAnalyzer:
111    """Ensures shadow doesn't write to canonical state."""
112
113    WRITE_KEYWORDS = {'assign', 'set', 'write', '=', 'store', 'put', 'push', 'pop'}
114
115    def analyze(self, module: ShadowModule) -> AnalysisResult:
116        critical_state_writes = 0
117        for keyword in self.WRITE_KEYWORDS:
118            if keyword in module.shadow_code.lower():
119                critical_state_writes += 1
120
121        # Specific checks: shadow should not contain 'canonical.' or 'canonical_state'
122        if 'canonical.' in module.shadow_code or 'canonical_state' in module.shadow_code:
123            return AnalysisResult(
124                analyzer="PlaneIsolation",
125                passed=False,
126                level=AnalysisLevel.CRITICAL,
127                message="Shadow block writes to canonical state"
128            )
129
130        return AnalysisResult(
131            analyzer="PlaneIsolation",
132            passed=True,
133            level=AnalysisLevel.CRITICAL,
134            message="Shadow block properly isolated from canonical state"
135        )

165class ResourceEstimator:
166    """Estimates CPU and memory usage of shadow block."""
167
168    CPU_LIMIT_MS = 1000
169    MEMORY_LIMIT_BYTES = 256 * 1024 * 1024  # 256MB
170
171    CPU_WEIGHTS = {
172        'for': 10, 'while': 20, 'sort': 50, 'map': 5, 'filter': 5,
173        'reduce': 10, 'recursion': 50, 'loop': 15
174    }
175
176    def analyze(self, module: ShadowModule) -> AnalysisResult:
177        code = module.shadow_code.lower()
178        estimated_cpu = 0
179        estimated_memory = len(code) * 2  # rough estimate per char
180
181        for keyword, weight in self.CPU_WEIGHTS.items():
182            count = code.count(keyword)
183            estimated_cpu += count * weight
184
185        # Base cost
186        estimated_cpu += len(code.split('\n')) * 5
187
188        issues = []
189        if estimated_cpu > self.CPU_LIMIT_MS:
190            issues.append(f"CPU estimate {estimated_cpu}ms exceeds {self.CPU_LIMIT_MS}ms limit")
191        if estimated_memory > self.MEMORY_LIMIT_BYTES:
192            issues.append(f"Memory estimate {estimated_memory} bytes exceeds {self.MEMORY_LIMIT_BYTES} bytes limit")
193
194        if issues:
195            return AnalysisResult(
196                analyzer="ResourceEstimator",
197                passed=False,
198                level=AnalysisLevel.CRITICAL if estimated_cpu > self.CPU_LIMIT_MS else AnalysisLevel.NON_CRITICAL,
199                message="; ".join(issues)
200            )
201
202        return AnalysisResult(
203            analyzer="ResourceEstimator",
204            passed=True,
205            level=AnalysisLevel.CRITICAL,
206            message=f"Resources within limits (CPU: ~{estimated_cpu}ms, Mem: ~{estimated_memory} bytes)"
207        )

234                level=AnalysisLevel.CRITICAL,
235                message=f"Impure operations in invariant: {', '.join(found_impure)}"
236            )
237
238        return AnalysisResult(
239            analyzer="PuritySpring",
240            passed=True,
241            level=AnalysisLevel.CRITICAL,
242            message="Invariant block is pure"

257        allocations += code.count('new') * 1000
258        allocations += code.count('list') * 500
259        allocations += code.count('map') * 500

266            return AnalysisResult(

272
273        return AnalysisResult(
274            analyzer="MemoryEvaporation",
275            passed=True,
276            level=AnalysisLevel.NON_CRITICAL,
277            message=f"Peak memory within limits (~{estimated_peak} bytes)"

281class CanonicalConvergenceAnalyzer:
282    """Ensures shadow and canonical blocks produce equivalent results."""
283
284    def analyze(self, module: ShadowModule) -> AnalysisResult:
285        if not module.shadow_code or not module.canonical_code:
286            return AnalysisResult(
287                analyzer="CanonicalConvergence",
288                passed=False,
289                level=AnalysisLevel.CRITICAL,
290                message="Both shadow and canonical blocks must be present"
291            )
292
293        # Check for structural convergence (both blocks exist and similar structure)
294        shadow_lines = [l.strip() for l in module.shadow_code.split('\n') if l.strip()]
295        canonical_lines = [l.strip() for l in module.canonical_code.split('\n') if l.strip()]
296
297        if len(shadow_lines) == 0 or len(canonical_lines) == 0:
298            return AnalysisResult(
299                analyzer="CanonicalConvergence",
300                passed=False,
301                level=AnalysisLevel.CRITICAL,
302                message="Shadow or canonical block is empty"
303            )
304
305        # Convergence check: Both should have similar structure
306        shadow_return_count = module.shadow_code.count('return')
307        canonical_return_count = module.canonical_code.count('return')
308
309        if shadow_return_count != canonical_return_count:
310            return AnalysisResult(
311                analyzer="CanonicalConvergence",
312                passed=False,
313                level=AnalysisLevel.NON_CRITICAL,
314                message=f"Shadow ({shadow_return_count} returns) and canonical ({canonical_return_count} returns) may not converge"
315            )
316
317        # Check output type consistency
318        shadow_ops = set(re.findall(r'\b[a-z_]+\b', module.shadow_code.lower()))
319        canonical_ops = set(re.findall(r'\b[a-z_]+\b', module.canonical_code.lower()))
320        overlap = shadow_ops & canonical_ops
321
322        if len(overlap) < max(1, min(len(shadow_ops), len(canonical_ops)) * 0.3):
323            return AnalysisResult(
324                analyzer="CanonicalConvergence",
325                passed=False,
326                level=AnalysisLevel.NON_CRITICAL,
327                message="Shadow and canonical blocks use very different operations — possible divergence"
328            )
329
330        return AnalysisResult(
331            analyzer="CanonicalConvergence",
332            passed=True,
333            level=AnalysisLevel.CRITICAL,
334            message="Shadow and canonical blocks converge"
335        )

338class PromotionEngine:
339    """Executes the promote/reject flow based on analyzer results."""
340
341    def __init__(self):
342        self.analyzers = [
343            ("PlaneIsolation", PlaneIsolationAnalyzer()),
344            ("Determinism", DeterminismAnalyzer()),
345            ("ResourceEstimation", ResourceEstimator()),
346            ("PuritySpring", PuritySpringAnalyzer()),
347            ("MemoryEvaporation", MemoryEvaporationAnalyzer()),
348            ("CanonicalConvergence", CanonicalConvergenceAnalyzer()),
349        ]
350
351    def evaluate(self, module: ShadowModule) -> Tuple[str, List[AnalysisResult]]:
352        """
353        Evaluate a mutation module. Returns (verdict, results).
354        Verdict is one of: PROMOTE, REJECT, FLAGGED
355        """
356        results = []
357
358        for name, analyzer in self.analyzers:
359            try:
360                result = analyzer.analyze(module)
361                results.append(result)
362            except Exception as e:
363                results.append(AnalysisResult(
364                    analyzer=name,
365                    passed=False,
366                    level=AnalysisLevel.CRITICAL,
367                    message=f"Analysis error: {e}"
368                ))
369
370        # Determine verdict
371        critical_failures = [r for r in results if not r.passed and r.level == AnalysisLevel.CRITICAL]
372        non_critical_failures = [r for r in results if not r.passed and r.level == AnalysisLevel.NON_CRITICAL]
373
374        if critical_failures:
375            verdict = "REJECT"
376        elif non_critical_failures:
377            verdict = "FLAGGED"
378        else:
379            verdict = "PROMOTE"
380
381        return verdict, results
382
383    @staticmethod
384    def generate_mermaid(module: ShadowModule, verdict: str = "PROMOTE", results: List[AnalysisResult] = None) -> str:
385        if results is None:
386            results = []
387        """Generate a Mermaid flowchart visualization of the promotion flow."""
388        lines = ["```mermaid", "flowchart TD"]
389        lines.append(f"    M[\"Mutation: {module.name}\"]")
390
391        for i, (name, _) in enumerate([
392            ("PlaneIsolationAnalyzer", None),
393            ("DeterminismAnalyzer", None),
394            ("ResourceEstimator", None),
395            ("PuritySpringAnalyzer", None),
396            ("MemoryEvaporationAnalyzer", None),
397            ("CanonicalConvergenceAnalyzer", None),
398        ]):
399            r = results[i] if i < len(results) else AnalysisResult(analyzer=name, passed=True)
400            status = "✅" if r.passed else "❌"
401            lines.append(f"    A{i}[\"{status} {name}\"]")
402
403            if i == 0:
404                lines.append(f"    M --> A{i}")
405            else:
406                lines.append(f"    A{i - 1} --> A{i}")
407
408        lines.append(f"    V[\"Verdict: {verdict}\"]")
409        lines.append(f"    A{min(5, len(results) - 1)} --> V")
410
411        if verdict == "PROMOTE":
412            lines.append(f"    V --> P[\"🚀 PROMOTE\"]")
413        elif verdict == "REJECT":
414            lines.append(f"    V --> R[\"❌ REJECT\"]")
415        else:
416            lines.append(f"    V --> F[\"⚠️ FLAGGED\"]")
417
418        lines.append("```")
419        return "\n".join(lines)

282class CanonicalConvergenceAnalyzer:
283    """Ensures shadow and canonical blocks produce equivalent results."""
284
285    def analyze(self, module: ShadowModule) -> AnalysisResult:
286        if not module.shadow_code or not module.canonical_code:
287            return AnalysisResult(
288                analyzer="CanonicalConvergence",
289                passed=False,
290                level=AnalysisLevel.CRITICAL,
291                message="Both shadow and canonical blocks must be present"
292            )
293
294        # Check for structural convergence (both blocks exist and similar structure)
295        shadow_lines = [l.strip() for l in module.shadow_code.split('\n') if l.strip()]
296        canonical_lines = [l.strip() for l in module.canonical_code.split('\n') if l.strip()]
297
298        if len(shadow_lines) == 0 or len(canonical_lines) == 0:
299            return AnalysisResult(
300                analyzer="CanonicalConvergence",
301                passed=False,
302                level=AnalysisLevel.CRITICAL,
303                message="Shadow or canonical block is empty"
304            )
305
306        # Convergence check: Both should have similar structure
307        shadow_return_count = module.shadow_code.count('return')
308        canonical_return_count = module.canonical_code.count('return')
309
310        if shadow_return_count != canonical_return_count:
311            return AnalysisResult(
312                analyzer="CanonicalConvergence",
313                passed=False,
314                level=AnalysisLevel.NON_CRITICAL,
315                message=f"Shadow ({shadow_return_count} returns) and canonical ({canonical_return_count} returns) may not converge"
316            )
317
318        # Check output type consistency
319        shadow_ops = set(re.findall(r'\b[a-z_]+\b', module.shadow_code.lower()))
320        canonical_ops = set(re.findall(r'\b[a-z_]+\b', module.canonical_code.lower()))
321        overlap = shadow_ops & canonical_ops
322
323        if len(overlap) < max(1, min(len(shadow_ops), len(canonical_ops)) * 0.3):
324            return AnalysisResult(
325                analyzer="CanonicalConvergence",
326                passed=False,
327                level=AnalysisLevel.NON_CRITICAL,
328                message="Shadow and canonical blocks use very different operations — possible divergence"
329            )
330
331        return AnalysisResult(
332            analyzer="CanonicalConvergence",
333            passed=True,
334            level=AnalysisLevel.CRITICAL,
335            message="Shadow and canonical blocks converge"
336        )

343        self.analyzers = [

353        """
354        Evaluate a mutation module. Returns (verdict, results).

374
375        if critical_failures:
376            verdict = "REJECT"
377        elif non_critical_failures:
378            verdict = "FLAGGED"
379        else:
380            verdict = "PROMOTE"
381
382        return verdict, results
